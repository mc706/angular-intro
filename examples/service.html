services/contact.js

angular.module('myApp')
    .service('ContactService', function ($http, $q) {
        return {
            listContacts: function () {
                var defer = $q.defer();
                $http({
                    method: "GET",
                    url: "/apis/contacts/"
                }).success(function (data, status) {
                    defer.resolve(data);
                }).error(function (data, status) {
                    defer.reject(status);
                });
                return defer.promise;
            },
            getContact: function (id) {
                var defer = $q.defer();
                $http({
                    method: "GET",
                    url: "/apis/contacts/" + id
                }).success(function (data, status) {
                    defer.resolve(data);
                }).error(function (data, status) {
                    defer.reject(status);
                });
                return defer.promise;
            },
            createContact: function (data) {
                var defer = $q.defer();
                $http({
                    method: "POST",
                    data: data
                    url: "/apis/contacts/"
                }).success(function (data, status) {
                    defer.resolve(data);
                }).error(function (data, status) {
                    defer.reject(status);
                });
                return defer.promise;
            },
            updateContact: function (id, data) {
                var defer = $q.defer();
                $http({
                    method: "PUT",
                    data: data
                    url: "/apis/contacts/" + id
                }).success(function (data, status) {
                    defer.resolve(data);
                }).error(function (data, status) {
                    defer.reject(status);
                });
                return defer.promise;
            }
            deleteContact: function (id) {
                var defer = $q.defer();
                $http({
                    method: "DELETE",
                    url: "/apis/contacts/" + id
                }).success(function (data, status) {
                    defer.resolve(data);
                }).error(function (data, status) {
                    defer.reject(status);
                });
                return defer.promise;
            }
        }
    });

controllers/myController.js

angular.module('myApp')
    .controller('MyController', function ($scope, ContactService) {
        $scope.contacts = ContactService.listContacts();

        $scope.deleteContact = function(id) {
            ContactService.deleteContact(id).then(function() {
                $scope.contacts = ContactService.listContacts();
            });
        }
    });
