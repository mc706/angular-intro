directives/contact-card/contact-card.js

angular.module('myApp')
    .directive('contactCard', function(){
        return {
            restrict: "E",
            replace: false,
            transclude: false,
            scope: {
                contact: "="
            },
            templateUrl: "directives/contact-card/card.html"

        }
    })

directives/contact-card/card.html

<div>
    <strong>{{contact.name}}</strong>
    <span ng-show='contact.phone'>
        {{contact.phone}}
    </span>
    <span ng-show='contact.email'>
        {{contact.email}}
    </span>
</div>

View

<div ng-controller="myController">

                  contact="contact">
    </contact-card>
</div>
